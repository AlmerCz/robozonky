97      /**
98       * Get the share of 'payments due for each rating' on the overall portfolio.
99       * @param stats
100      * @param investments Loans which have already been invested in by the current user.
101      * @return Map where each rating is the key and value is the share of that rating among overall due payments.
102      */
103     static Map<Rating, BigDecimal> calculateSharesPerRating(final Statistics stats,
104                                                             final Collection<Investment> investments) {
105         final Map<Rating, BigDecimal> amounts = stats.getRiskPortfolio().stream().collect(
106                 Collectors.toMap(RiskPortfolio::getRating, risk -> BigDecimal.valueOf(risk.getUnpaid()))
107         );
108         // make sure ratings are present even when there's 0 invested in them
109         Arrays.stream(Rating.values()).filter(r -> !amounts.containsKey(r))
110                 .forEach(r -> amounts.put(r, BigDecimal.ZERO));
111         // make sure the share reflects investments made by ZonkyBot which have not yet been reflected in the API
112         investments.forEach(previousInvestment -> {
113             final Rating r = previousInvestment.getRating();
114             final BigDecimal investment = BigDecimal.valueOf(previousInvestment.getAmount());
115             amounts.put(r, amounts.get(r).add(investment));
116         });
117         final BigDecimal total = Util.sum(amounts.values());
118         if (total.compareTo(BigDecimal.ZERO) == 0) { // no ratings have any investments
119             return Collections.unmodifiableMap(amounts);
120         }
121         final Map<Rating, BigDecimal> result = new EnumMap<>(Rating.class);
122         amounts.forEach((rating, amount) -> result.put(rating, amount.divide(total, 4, RoundingMode.HALF_EVEN)));
123         return Collections.unmodifiableMap(result);
124     }

126     /**
127      * Put money into an already selected loan.
128      * @param oc Context for the current session.
129      * @param l Loan to invest into.
130      * @param investmentsInSession Previous loans invested in this session.
131      * @param balance Latest known Zonky account balance.
132      * @return Present only if Zonky API confirmed money was invested or if dry run.
133      */
134     static Optional<Investment> actuallyInvest(final OperationsContext oc, final Loan l,
135                                                final List<Investment> investmentsInSession, final BigDecimal balance) {
136         if (Util.isLoanPresent(l, investmentsInSession)) {
137             Operations.LOGGER.info("RoboZonky already invested in loan '{}', skipping. May only happen in dry runs.", l);
138             return Optional.empty();
139         } else if (!oc.getStrategy().isAcceptable(l)) {
140             Operations.LOGGER.info("According to the investment strategy, loan '{}' is not acceptable.", l);
141             return Optional.empty();
142         }
143         // figure out how much to invest
144         final int resultingInvestment = oc.getStrategy().recommendInvestmentAmount(l, balance);
145         Operations.LOGGER.debug("Strategy recommended to invest {} CZK on balance of {} CZK.", resultingInvestment,
146                 balance.intValue());
147         if (resultingInvestment < Operations.MINIMAL_INVESTMENT_ALLOWED) {
148             Operations.LOGGER.info("Not investing into loan '{}', since investment ({} CZK) less than bare minimum.",
149                     l, resultingInvestment);
150             return Optional.empty();
151         } else if (resultingInvestment > balance.intValue()) {
152             Operations.LOGGER.info("Not investing into loan '{}', since investment ({} CZK) more than {} CZK balance.",
153                     l, resultingInvestment, balance);
154             return Optional.empty();
155         }
156         return Operations.invest(oc, l, resultingInvestment);
157     }

159     /**
160      * Choose from available loans of a given rating one loan to invest money into.
161      * @param oc Context for the current session.
162      * @param r Rating in question.
163      * @param loansFuture Loans carrying that rating.
164      * @param investmentsInSession Previous investments made in this session.
165      * @param balance Latest known Zonky account balance.
166      * @return Present only if Zonky API confirmed money was invested or if dry run.
167      */
168     static Optional<Investment> identifyLoanToInvest(final OperationsContext oc, final Rating r,
169                                                      final Future<Collection<Loan>> loansFuture,
170                                                      final List<Investment> investmentsInSession,
171                                                      final BigDecimal balance) {
172         final Collection<Loan> loans;
173         try {
174             loans = loansFuture.get();
175         } catch (final Exception e) {
176             Operations.LOGGER.warn("Could not list loans with rating '{}'. Can not invest in that rating.", r, e);
177             return Optional.empty();
178         }
179         if (loans.size() == 0) {
180             Operations.LOGGER.info("There are no loans of rating '{}' matching the investment strategy.", r);
181             return Optional.empty();
182         }
183         // sort loans by their term and start investing
184         final boolean prefersLongTerm = oc.getStrategy().prefersLongerTerms(r);
185         Operations.LOGGER.info("Investment strategy for rating '{}' prefers {} term loans.", r,
186                 prefersLongTerm ? "longer" : "shorter");
187         for (final Loan l : Util.sortLoansByTerm(loans, prefersLongTerm)) {
188             final Optional<Investment> investment = Operations.actuallyInvest(oc, l, investmentsInSession, balance);
189             if (investment.isPresent()) {
190                 return investment;
191             }
192         }
193         return Optional.empty();
194     }

196     /**
197      *
198      * @param oc Context for the current session.
199      * @param currentShare Current share of investments in a given rating.
200      * @return Ratings in the order of decreasing demand. Over-invested ratings not present.
201      */
202     static List<Rating> rankRatingsByDemand(final OperationsContext oc, final Map<Rating, BigDecimal> currentShare) {
203         final MultiValuedMap<BigDecimal, Rating> mostWantedRatings = new HashSetValuedHashMap<>();
204         // put the ratings into buckets based on how much we're missing them
205         currentShare.forEach((r, currentRatingShare) -> {
206             final BigDecimal maximumAllowedShare = oc.getStrategy().getTargetShare(r);
207             if (currentRatingShare.compareTo(maximumAllowedShare) >= 0) { // we bought too many of this rating; ignore
208                 return;
209             }
210             mostWantedRatings.put(maximumAllowedShare.subtract(currentRatingShare), r);
211         });
212         // and now output ratings in an order, bigger first
213         final List<Rating> result = new ArrayList<>(currentShare.size());
214         while (!mostWantedRatings.isEmpty()) {
215             BigDecimal biggestRanking = BigDecimal.ZERO;
216             for (final BigDecimal tested: mostWantedRatings.keySet()) {
217                 if (tested.compareTo(biggestRanking) > 0) {
218                     biggestRanking = tested;
219                 }
220             }
221             result.addAll(mostWantedRatings.remove(biggestRanking));
222         }
223         return Collections.unmodifiableList(result);
224     }

226     /**
227      * Choose from available loans the most important loan to invest money into.
228      * @param oc Context for the current session.
229      * @param investmentsInSession Previous investments made in this session.
230      * @param balance Latest known Zonky account balance.
231      * @return Present only if Zonky API confirmed money was invested or if dry run.
232      */
233     static Optional<Investment> identifyLoanToInvest(final OperationsContext oc,
234                                                      final List<Investment> investmentsInSession,
235                                                      final BigDecimal balance) {
236         final ZonkyApi api = oc.getApi();
237         final Collection<Investment> investments = Util.mergeInvestments(
238                 api.getInvestments(InvestmentStatuses.of(InvestmentStatus.SIGNED)), investmentsInSession);
239         final Map<Rating, BigDecimal> shareOfRatings = Operations.calculateSharesPerRating(api.getStatistics(),
240                 investments);
241         final List<Rating> mostWantedRatings = Operations.rankRatingsByDemand(oc, shareOfRatings);
242         Operations.LOGGER.debug("Current share of unpaid loans with a given rating is currently: {}.", shareOfRatings);
243         Operations.LOGGER.info("According to the investment strategy, the portfolio is low on following ratings: {}.",
244                 mostWantedRatings);
245         final Map<Rating, Future<Collection<Loan>>> availableLoans = new EnumMap<>(Rating.class);
246         mostWantedRatings.forEach(r -> { // submit HTTP requests ahead of time
247             final Callable<Collection<Loan>> future
248                     = () -> api.getLoans(Ratings.of(r), Operations.MINIMAL_INVESTMENT_ALLOWED);
249             availableLoans.put(r, oc.getBackgroundExecutor().submit(future));
250         });
251         for (final Rating r : mostWantedRatings) { // try to invest in a given rating
252             final Optional<Investment> investment =
253                     Operations.identifyLoanToInvest(oc, r, availableLoans.get(r), investmentsInSession, balance);
254             if (investment.isPresent()) {
255                 return investment;
256             }
257         }
258         return Optional.empty();
259     }

261     static BigDecimal getAvailableBalance(final OperationsContext oc,
262                                           final Collection<Investment> investmentsInSession) {
263         final boolean isDryRun = oc.isDryRun();
264         final BigDecimal dryRunInitialBalance = oc.getDryRunInitialBalance();
265         BigDecimal balance = (isDryRun && dryRunInitialBalance.compareTo(BigDecimal.ZERO) > 0) ?
266                 dryRunInitialBalance : oc.getApi().getWallet().getAvailableBalance();
267         if (isDryRun) {
268             for (final Investment i : investmentsInSession) {
269                 balance = balance.subtract(BigDecimal.valueOf(i.getAmount()));
270             }
271         }
272         return balance;
273     }

275     public static Collection<Investment> invest(final OperationsContext oc) {
276         final int minimumInvestmentAmount = Operations.MINIMAL_INVESTMENT_ALLOWED;
277         final List<Investment> investmentsMade = new ArrayList<>();
278         BigDecimal availableBalance = Operations.getAvailableBalance(oc, investmentsMade);
279         Operations.LOGGER.info("RoboZonky starting account balance is {} CZK.", availableBalance);
280         while (availableBalance.compareTo(BigDecimal.valueOf(minimumInvestmentAmount)) >= 0) {
281             final Optional<Investment> investment =
282                     Operations.identifyLoanToInvest(oc, investmentsMade, availableBalance);
283             if (investment.isPresent()) {
284                 investmentsMade.add(investment.get());
285                 availableBalance = Operations.getAvailableBalance(oc, investmentsMade);
286                 if (oc.isDryRun()) {
287                     Operations.LOGGER.info("Simulated new account balance is {} CZK.", availableBalance);
288                 } else {
289                     Operations.LOGGER.info("New account balance is {} CZK.", availableBalance);
290                 }
291             } else {
292                 break;
293             }
294         }
295         return Collections.unmodifiableList(investmentsMade);
296     }

298     private static Optional<Investment> invest(final OperationsContext oc, final Investment i) {
299         if (oc.isDryRun()) {
300             Operations.LOGGER.info("This is a dry run. Not investing {} CZK into loan {}.", i.getAmount(),
301                     i.getLoanId());
302             return Optional.of(i);
303         } else {
304             Operations.LOGGER.info("Attempting to invest {} CZK into loan {}.", i.getAmount(), i.getLoanId());
305             try {
306                 oc.getApi().invest(i);
307                 Operations.LOGGER.warn("Investment operation succeeded.");
308                 return Optional.of(i);
309             } catch (final Exception ex) {
310                 Operations.LOGGER.warn("Investment operation failed.", ex);
311                 return Optional.empty();
312             }
313         }
314     }

326     public static LoginResult login(final Authenticator authenticationMethod, final boolean dryRun,
327                                     final int dryRunInitialBalance, final InvestmentStrategy strategy)
328             throws LoginFailedException {
329         try {
330             Operations.LOGGER.trace("Preparing for login.");
331             // authenticate
332             final ResteasyClientBuilder clientBuilder = new ResteasyClientBuilder();
333             clientBuilder.providerFactory(Operations.RESTEASY);
334             clientBuilder.connectionPoolSize(Operations.CONNECTION_POOL_SIZE);
335             Operations.LOGGER.trace("Login starting.");
336             final Authentication auth =
337                     authenticationMethod.authenticate(Operations.ZONKY_URL, Util.getRoboZonkyVersion(), clientBuilder);
338             final OperationsContext oc = new OperationsContext(auth.getApi(), strategy, dryRun, dryRunInitialBalance,
339                     Operations.CONNECTION_POOL_SIZE);
340             return new LoginResult(oc, auth.getApiToken());
341         } catch (final RuntimeException ex) {
342             throw new LoginFailedException("Error while instantiating Zonky API proxy.", ex);
343         } finally {
344             Operations.LOGGER.trace("Login complete.");
345         }
346     }

353     public static void logout(final OperationsContext oc) throws LogoutFailedException {
354         try {
355             oc.getApi().logout();
356         } catch (final RuntimeException ex) {
357             throw new LogoutFailedException("Error while logging out Zonky.", ex);
358         }
359         oc.dispose();
360         Operations.LOGGER.info("Logged out of Zonky.");
361     }
