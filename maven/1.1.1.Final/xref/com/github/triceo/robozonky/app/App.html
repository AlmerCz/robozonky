52      static void exit(final ReturnCode returnCode) {
53          if (App.PERFORM_SYSTEM_EXIT) {
54              App.LOGGER.debug("RoboZonky terminating with '{}' return code.", returnCode);
55              System.exit(returnCode.getCode());
56          } else {
57              throw new RoboZonkyTestingExitException("Return code: " + returnCode);
58          }
59      }

61      private static AppContext prepareStrategyDrivenMode(final CommandLineInterface cli) {
62          final Optional<Integer> loanAmount = cli.getLoanAmount();
63          if (loanAmount.isPresent()) {
64              cli.printHelpAndExit("Loan amount makes no sense in this context.", true);
65          }
66          final Optional<String> strategyFilePath = cli.getStrategyConfigurationFilePath();
67          if (!strategyFilePath.isPresent()) {
68              cli.printHelpAndExit("Strategy file must be provided.", true);
69          }
70          final File strategyConfig = new File(strategyFilePath.get());
71          if (!strategyConfig.canRead()) {
72              cli.printHelpAndExit("Investment strategy file must be readable.", true);
73          }
74          final AuthenticationHandler auth = App.getAuthenticationMethod(cli);
75          try {
76              if (cli.isDryRun()) {
77                  return new AppContext(auth, StrategyParser.parse(strategyConfig), cli.getDryRunBalance());
78              } else {
79                  return new AppContext(auth, StrategyParser.parse(strategyConfig));
80              }
81          } catch (final Exception ex) {
82              cli.printHelpAndExit("Failed parsing strategy.", ex);
83              return null;
84          }
85      }

87      private static SensitiveInformationProvider getSensitiveInformationProvider(final CommandLineInterface cli) {
88          final Optional<File> keyStoreLocation = cli.getKeyStoreLocation();
89          if (keyStoreLocation.isPresent()) { // if user requests keystore, cli is only used to retrieve keystore file
90              final File store = keyStoreLocation.get();
91              try {
92                  final KeyStoreHandler ksh = KeyStoreHandler.open(store, cli.getPassword());
93                  return SensitiveInformationProvider.keyStoreBased(ksh);
94              } catch (final IOException | KeyStoreException ex) {
95                  cli.printHelpAndExit("Failed opening guarded storage.", ex);
96                  return null;
97              }
98          } else { // else everything is read from the cli and put into a keystore
99              try {
100                 final Optional<String> usernameProvided = cli.getUsername();
101                 final boolean usernamePresent = usernameProvided.isPresent();
102                 final boolean storageExists = App.DEFAULT_KEYSTORE_FILE.canRead();
103                 if (storageExists) {
104                     if (App.DEFAULT_KEYSTORE_FILE.delete()) {
105                         App.LOGGER.debug("Deleted pre-existing guarded storage.");
106                     } else {
107                         throw new IllegalArgumentException("Stale guarded storage is present and can not be deleted.");
108                     }
109                 }
110                 final KeyStoreHandler ksh = KeyStoreHandler.create(App.DEFAULT_KEYSTORE_FILE, cli.getPassword());
111                 if (!usernamePresent) {
112                     cli.printHelpAndExit("When not using guarded storage, username must be provided.", true);
113                 } else if (storageExists) {
114                     App.LOGGER.info("Using plain-text credentials when guarded storage available. Consider switching.");
115                 } else {
116                     App.LOGGER.info("Guarded storage has been created with your username and password: {}",
117                             App.DEFAULT_KEYSTORE_FILE);
118                     App.LOGGER.info("Feel free to use this instead of providing the information on the command line.");
119                     App.LOGGER.info("Please change the storage password to something else than your Zonky password.");
120                 }
121                 return SensitiveInformationProvider.keyStoreBased(ksh, usernameProvided.get(), cli.getPassword());
122             } catch (final IOException | KeyStoreException ex) {
123                 cli.printHelpAndExit("Failed reading guarded storage.", ex);
124                 return null;
125             }
126         }
127     }

129     private static AuthenticationHandler getAuthenticationMethod(final CommandLineInterface cli) {
130         final boolean useToken = cli.isTokenEnabled();
131         final SensitiveInformationProvider sensitive = App.getSensitiveInformationProvider(cli);
132         final AuthenticationHandler auth = useToken ? AuthenticationHandler.tokenBased(sensitive)
133                 : AuthenticationHandler.passwordBased(sensitive);
134         final Optional<Integer> secs = cli.getTokenRefreshBeforeExpirationInSeconds();
135         if (secs.isPresent()) {
136             auth.withTokenRefreshingBeforeExpiration(secs.get(), ChronoUnit.SECONDS);
137         }
138         return auth;
139     }

141     private static AppContext prepareUserDrivenMode(final CommandLineInterface cli) {
142         final Optional<Integer> loanId = cli.getLoanId();
143         final Optional<Integer> loanAmount = cli.getLoanAmount();
144         if (!loanId.isPresent()) {
145             cli.printHelpAndExit("Loan ID must be provided.", true);
146         } else if (!loanAmount.isPresent()) {
147             cli.printHelpAndExit("Loan amount must be provided.", true);
148         }
149         final AuthenticationHandler auth = App.getAuthenticationMethod(cli);
150         if (cli.isDryRun()) {
151             return new AppContext(auth, loanId.get(), loanAmount.get(), cli.getDryRunBalance());
152         } else {
153             return new AppContext(auth, loanId.get(), loanAmount.get());
154         }
155     }

157     private static AppContext processCommandLine(final String... args) {
158         final CommandLineInterface cli = CommandLineInterface.parse(args);
159         final Optional<OperatingMode> om = cli.getCliOperatingMode();
160         if (!om.isPresent()) {
161             cli.printHelpAndExit("", false);
162             return null;
163         }
164         switch (om.get()) {
165             case STRATEGY_DRIVEN:
166                 return App.prepareStrategyDrivenMode(cli);
167             case USER_DRIVER:
168                 return App.prepareUserDrivenMode(cli);
169             default:
170                 return null;
171         }
172     }

174     public static void main(final String... args) {
175         App.LOGGER.info("RoboZonky v{} loading.", Util.getRoboZonkyVersion());
176         try {
177             App.letsGo(App.processCommandLine(args)); // and start actually working with Zonky
178             App.exit(ReturnCode.OK);
179         } catch (final Exception ex) {
180             App.LOGGER.error("Unexpected error." , ex);
181             App.exit(ReturnCode.ERROR_UNEXPECTED);
182         }
183     }

185     private static void storeInvestmentsMade(final Collection<Investment> result, final boolean dryRun) {
186         final String suffix = dryRun ? "dry" : "invested";
187         final LocalDateTime now = LocalDateTime.now();
188         final String filename =
189                 "robozonky." + DateTimeFormatter.ofPattern("yyyyMMddHHmm").format(now) + '.' + suffix;
190         final File target = new File(filename);
191         try (final BufferedWriter bw = Files.newBufferedWriter(target.toPath(), Charset.forName("UTF-8"))) {
192             for (final Investment i : result) {
193                 bw.write('#' + i.getLoanId() + ": " + i.getAmount() + " CZK");
194                 bw.newLine();
195             }
196             App.LOGGER.info("Investments made by RoboZonky during the session were stored in file '{}'.", filename);
197         } catch (final IOException ex) {
198             App.LOGGER.warn("Failed writing out the list of investments made in this session.", ex);
199         }
200     }

202     private static void letsGo(final AppContext ctx) {
203         App.LOGGER.info("===== RoboZonky at your service! =====");
204         final boolean dryRun = ctx.isDryRun();
205         if (dryRun) {
206             App.LOGGER.info("RoboZonky is doing a dry run. It will simulate investing, but not invest any real money.");
207         }
208         // figure out whether to invest based on strategy or whether to make a single investment
209         final boolean useStrategy = ctx.getOperatingMode() == OperatingMode.STRATEGY_DRIVEN;
210         final Function<OperationsContext, Collection<Investment>> op = useStrategy ? Operations::invest : oc -> {
211             final Optional<Investment> optional = Operations.invest(oc, ctx.getLoanId(), ctx.getLoanAmount());
212             if (optional.isPresent()) {
213                 return Collections.singletonList(optional.get());
214             } else {
215                 return Collections.emptyList();
216             }
217         };
218         // and now perform the selected operation
219         final Collection<Investment> result = App.operate(ctx, op);
220         if (result.isEmpty()) {
221             App.LOGGER.info("RoboZonky did not invest.");
222         } else {
223             App.storeInvestmentsMade(result, dryRun);
224             if (dryRun) {
225                 App.LOGGER.info("RoboZonky pretended to invest into {} loans.", result.size());
226             } else {
227                 App.LOGGER.info("RoboZonky invested into {} loans.", result.size());
228             }
229         }
230         App.LOGGER.info("===== RoboZonky out. =====");
231     }

233     private static Collection<Investment> operate(final AppContext ctx,
234                                                   final Function<OperationsContext, Collection<Investment>> operations) {
235         final boolean useStrategy = ctx.getOperatingMode() == OperatingMode.STRATEGY_DRIVEN;
236         try {
237             final AuthenticationHandler handler = ctx.getAuthenticationHandler();
238             final Authenticator auth = handler.build();
239             final Operations.LoginResult loginResult = useStrategy ?
240                     Operations.login(auth, ctx.isDryRun(), ctx.getDryRunBalance(), ctx.getInvestmentStrategy()) :
241                     Operations.login(auth, ctx.isDryRun(), ctx.getDryRunBalance());
242             final boolean logoutAllowed = handler.processToken(loginResult.getZonkyApiToken());
243             final Collection<Investment> result = operations.apply(loginResult.getOperationsContext());
244             if (logoutAllowed) { // log out
245                 try {
246                     Operations.logout(loginResult.getOperationsContext());
247                 } catch (final LogoutFailedException ex) {
248                     App.LOGGER.warn("Logging out of Zonky failed.", ex);
249                 }
250             } else {  // if we're using the token, we should never log out
251                 App.LOGGER.info("Refresh token needs to be reused, not logging out of Zonky.");
252             }
253             return result;
254         } catch (final LoginFailedException ex) {
255             App.LOGGER.error("Logging into Zonky failed. No investments were made.", ex);
256             App.exit(ReturnCode.ERROR_LOGIN);
257         }
258         return Collections.emptyList(); // should never get here
259     }
