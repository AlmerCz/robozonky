17  package com.github.triceo.robozonky.remote;
18  
19  import org.assertj.core.api.Assertions;
20  import org.junit.Test;
21  
22  public class RatingsTest {
23  
24      @Test
25      public void ratingsAreProperlySerialized() {
26          Assertions.assertThat(Ratings.of(Rating.A).toString()).isEqualTo("[\"A\"]");
27          Assertions.assertThat(Ratings.of(Rating.AA, Rating.AAAA, Rating.AA).toString()).isEqualTo("[\"AAAA\", \"AA\"]");
28      }
29  
30      @Test
31      public void ratingsAreProperlyEnumerated() {
32          Assertions.assertThat(Ratings.of(Rating.AAA).getRatings()).containsExactly(Rating.AAA);
33          Assertions.assertThat(Ratings.of(Rating.B, Rating.C, Rating.B).getRatings()).containsExactly(Rating.B, Rating.C);
34      }
35  
36      @Test
37      public void ofAll() {
38          final Ratings result = Ratings.all();
39          Assertions.assertThat(result.getRatings()).containsOnly(Rating.values());
40      }
41  
42      @Test
43      public void correctValueOf() {
44          // test no items
45          Assertions.assertThat(Ratings.valueOf("[]").getRatings()).isEmpty();
46          Assertions.assertThat(Ratings.valueOf(" [ ] ").getRatings()).isEmpty();
47          // test one item
48          Assertions.assertThat(Ratings.valueOf("[ \"A\" ]").getRatings())
49                  .containsExactly(Rating.A);
50          Assertions.assertThat(Ratings.valueOf(" [ \"AA\"]").getRatings())
51                  .containsExactly(Rating.AA);
52          // test multiple items
53          Assertions.assertThat(Ratings.valueOf(" [\"AAA\", \"B\"]").getRatings())
54                  .containsExactly(Rating.AAA, Rating.B);
55          Assertions.assertThat(Ratings.valueOf(" [\"B\" , \"AAA\"] ").getRatings())
56                  .containsExactly(Rating.AAA, Rating.B);
57      }
58  
59      @Test(expected = IllegalArgumentException.class)
60      public void invalidValueOf() {
61          Ratings.valueOf("[");
62      }
63  
64      @Test(expected = IllegalArgumentException.class)
65      public void unquotedValueOf() {
66          Ratings.valueOf("[A]");
67      }
68  
69      @Test(expected = IllegalArgumentException.class)
70      public void unknownValueOf() {
71          Ratings.valueOf("[\"SOME_UNKNOWN_VALUE\"]");
72      }
73  }
