40  /**
41   * This strategy implements evaluation using a Drools decision table. See http://www.drools.org/
42   *
43   * Before {@link #recommendInvestmentAmount(Loan, PortfolioOverview)}, you must call
44   * {@link #getMatchingLoans(List, PortfolioOverview)}. Otherwise the strategy does not have the decision data.
45   */
46  class RuleBasedInvestmentStrategy implements InvestmentStrategy {
47  
48      private static final Logger LOGGER = LoggerFactory.getLogger(RuleBasedInvestmentStrategy.class);
49  
50      private static Loan matchLoan(final AcceptedLoan l, final List<Loan> loans) {
51          for (final Loan loan: loans) {
52              if (loan.getId() == l.getId()) {
53                  return loan;
54              }
55          }
56          throw new IllegalStateException("Could not find matching loan. Should not have happened.");
57      }
58  
59      private final KieContainer kieContainer;
60      private final Map<Loan, Integer> recommendedAmounts = new HashMap<>();
61  
62      RuleBasedInvestmentStrategy(final KieContainer kieContainer) {
63          this.kieContainer = kieContainer;
64      }
65  
66      @Override
67      @SuppressWarnings("unchecked")
68      public synchronized List<Loan> getMatchingLoans(final List<Loan> availableLoans,
69                                                      final PortfolioOverview portfolio) {
70          this.recommendedAmounts.clear();
71          RuleBasedInvestmentStrategy.LOGGER.trace("Started matching loans.");
72          final KieSession session = this.kieContainer.newKieSession();
73          // insert facts into session
74          availableLoans.stream().map(ProposedLoan::new).forEach(session::insert);
75          Arrays.stream(Rating.values())
76                  .map(r -> new RatingShare(r, portfolio.getShareOnInvestment(r)))
77                  .forEach(session::insert);
78          session.insert(new Wallet(portfolio.getCzkAvailable(), portfolio.getCzkInvested()));
79          // reason over facts and process results
80          RuleBasedInvestmentStrategy.LOGGER.trace("Facts inserted into session.");
81          session.fireAllRules();
82          RuleBasedInvestmentStrategy.LOGGER.trace("Drools finished.");
83          final Collection<AcceptedLoan> result =
84                  (Collection<AcceptedLoan>)session.getObjects(o -> o instanceof AcceptedLoan);
85          session.dispose();
86          RuleBasedInvestmentStrategy.LOGGER.trace("Session disposed.");
87          final Map<AcceptedLoan, Loan> map = result.stream()
88                  .collect(Collectors.toMap(Function.identity(),
89                          l -> RuleBasedInvestmentStrategy.matchLoan(l, availableLoans)));
90          map.forEach((fake, actual) -> this.recommendedAmounts.put(actual, fake.getAmount()));
91          RuleBasedInvestmentStrategy.LOGGER.trace("Found recommended amounts.");
92          // return results in the order of decreasing priority
93          final List<Loan> finalResult = result.stream()
94                  .sorted((a, b) -> Integer.compare(b.getPriority(), a.getPriority()))
95                  .map(map::get)
96                  .collect(Collectors.toList());
97          RuleBasedInvestmentStrategy.LOGGER.trace("Loans matched.");
98          return finalResult;
99      }
100 
101     /**
102      * Does not actually do anything. Only returns a result that had previously been calculated when
103      * {@link #getMatchingLoans(List, PortfolioOverview)} was called.
104      *
105      * @param loan Loan in question.
106      * @param portfolio Aggregation of information as to the user's current portfolio.
107      * @return How much should be invested into the loan.
108      */
109     @Override
110     public synchronized int recommendInvestmentAmount(final Loan loan, final PortfolioOverview portfolio) {
111         final Integer result = this.recommendedAmounts.get(loan);
112         return (result == null) ? 0 : result;
113     }
114 
115 }
