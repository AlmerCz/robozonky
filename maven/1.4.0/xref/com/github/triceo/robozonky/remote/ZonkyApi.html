17  package com.github.triceo.robozonky.remote;
18  
19  import java.util.Collection;
20  import java.util.List;
21  import javax.ws.rs.DefaultValue;
22  import javax.ws.rs.FormParam;
23  import javax.ws.rs.GET;
24  import javax.ws.rs.HeaderParam;
25  import javax.ws.rs.POST;
26  import javax.ws.rs.Path;
27  import javax.ws.rs.PathParam;
28  import javax.ws.rs.QueryParam;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  public interface ZonkyApi extends Api {
34  
35      Logger LOGGER = LoggerFactory.getLogger(ZonkyApi.class);
36  
37      String LOANS = "/loans";
38      String MARKETPLACE = LOANS + "/marketplace";
39      String ME = "/users/me";
40      String WALLET = ME + "/wallet";
41      String INVESTMENTS = ME + "/investments";
42      String OAUTH = "/oauth/token";
43  
44      @POST
45      @Path(ZonkyApi.OAUTH)
46      ZonkyApiToken login(
47              @FormParam("username") String username,
48              @FormParam("password") String password,
49              @FormParam("grant_type") @DefaultValue("password") String grantType,
50              @FormParam("scope") @DefaultValue("SCOPE_APP_WEB") String scope);
51  
52      @POST
53      @Path(ZonkyApi.OAUTH)
54      ZonkyApiToken refresh(
55              @FormParam("refresh_token") String refreshToken,
56              @FormParam("grant_type") @DefaultValue("refresh_token") String grantType,
57              @FormParam("scope") @DefaultValue("SCOPE_APP_WEB") String scope);
58  
59      @GET
60      @Path(ZonkyApi.WALLET)
61      Wallet getWallet();
62  
63      @GET
64      @Path(ZonkyApi.WALLET + "/blocked-amounts")
65      List<BlockedAmount> getBlockedAmounts(@HeaderParam("X-Size") int pageSize, @HeaderParam("X-Page") int pageNo);
66  
67      @GET
68      @Path(ZonkyApi.LOANS + "/{loanId}")
69      Loan getLoan(@PathParam("loanId") int loanId);
70  
71      @GET
72      @Path(ZonkyApi.MARKETPLACE)
73      List<Loan> getLoans(
74              @QueryParam("rating__in") Ratings ratings,
75              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment,
76              @QueryParam("termInMonths__gte") @DefaultValue("0") int leastPossibleTermInMonths,
77              @QueryParam("termInMonths__lte") int mostPossibleTermInMonths);
78  
79      @GET
80      @Path(ZonkyApi.MARKETPLACE)
81      List<Loan> getLoans(
82              @QueryParam("rating__in") Ratings ratings,
83              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment,
84              @QueryParam("termInMonths__gte") @DefaultValue("0") int leastPossibleTermInMonths);
85  
86      @GET
87      @Path(ZonkyApi.MARKETPLACE)
88      List<Loan> getLoans(
89              @QueryParam("rating__in") Ratings ratings,
90              @QueryParam("remainingInvestment__gt") @DefaultValue("0") int leastRemainingInvestment);
91  
92      @GET
93      @Path(ZonkyApi.MARKETPLACE)
94      List<Loan> getLoans(@QueryParam("rating__in") Ratings ratings);
95  
96      @GET
97      @Path(ZonkyApi.MARKETPLACE)
98      @Override
99      List<Loan> getLoans();
100 
101 
102     @GET
103     @Path(ZonkyApi.ME + "/logout")
104     void logout();
105 
106     @GET
107     @Path(ZonkyApi.INVESTMENTS + "/statistics")
108     Statistics getStatistics();
109 
110     @GET
111     @Path(ZonkyApi.INVESTMENTS)
112     Collection<Investment> getInvestments(
113             @QueryParam("loan.status__in") InvestmentStatuses statuses,
114             @QueryParam("loan.dpd__gt") @DefaultValue("0") int daysPastDueGreaterThan);
115 
116     @GET
117     @Path(ZonkyApi.INVESTMENTS)
118     Collection<Investment> getInvestments(@QueryParam("loan.status__in") InvestmentStatuses statuses);
119 
120 }
