88      @Test
89      public void properlyRecommendingInvestmentSize() {
90          final BigDecimal loanAmount = BigDecimal.valueOf(100000.0);
91          final Loan mockLoan = Mockito.mock(Loan.class);
92          Mockito.when(mockLoan.getRating()).thenReturn(SimpleInvestmentStategyTest.RATING_A);
93          Mockito.when(mockLoan.getRemainingInvestment()).thenReturn(loanAmount.doubleValue());
94          Mockito.when(mockLoan.getAmount()).thenReturn(loanAmount.doubleValue());
95  
96          // with unlimited balance
97          final PortfolioOverview portfolio = Mockito.mock(PortfolioOverview.class);
98          Mockito.when(portfolio.getCzkAvailable()).thenReturn(Integer.MAX_VALUE);
99          Mockito.when(portfolio.getCzkInvested()).thenReturn(0);
100         final int maxInvestment = Math.min(
101                 loanAmount.multiply(SimpleInvestmentStategyTest.MAXIMUM_LOAN_SHARE_A).intValue(),
102                 SimpleInvestmentStategyTest.MAXIMUM_LOAN_INVESTMENT);
103         final int actualInvestment = overallStrategy.recommendInvestmentAmount(mockLoan, portfolio);
104         Assertions.assertThat(actualInvestment).isLessThanOrEqualTo(maxInvestment);
105 
106         // with balance just a little less than the recommended investment
107         Mockito.when(portfolio.getCzkAvailable()).thenReturn(actualInvestment - 1);
108         final int adjustedForBalance = overallStrategy.recommendInvestmentAmount(mockLoan, portfolio);
109         Assertions.assertThat(adjustedForBalance).isLessThanOrEqualTo(actualInvestment);
110 
111         // and make sure the recommendation is 200 times X, where X is a positive integer
112         final BigDecimal remainder = BigDecimal.valueOf(adjustedForBalance)
113                 .remainder(BigDecimal.valueOf(InvestmentStrategy.MINIMAL_INVESTMENT_INCREMENT));
114         Assertions.assertThat(remainder.intValue()).isEqualTo(0);
115     }

117     private static Map<Rating, BigDecimal> prepareShareMap(final BigDecimal ratingA, final BigDecimal ratingB,
118                                                            final BigDecimal ratingC) {
119         final Map<Rating, BigDecimal> map = new EnumMap<>(Rating.class);
120         map.put(Rating.A, ratingA);
121         map.put(Rating.B, ratingB);
122         map.put(Rating.C, ratingC);
123         return Collections.unmodifiableMap(map);
124     }

126     private static void assertOrder(final List<Rating> result, final Rating... ratingsOrderedDown) {
127         Assertions.assertThat(result).hasSize(ratingsOrderedDown.length);
128         if (ratingsOrderedDown.length < 2) {
129             return;
130         } else if (ratingsOrderedDown.length > 3) {
131             throw new IllegalStateException("This should never happen in the test.");
132         }
133         final Rating first = result.get(0);
134         final Rating last = result.get(result.size() - 1);
135         Assertions.assertThat(first).isGreaterThan(last);
136         Assertions.assertThat(first).isEqualTo(ratingsOrderedDown[0]);
137         Assertions.assertThat(last).isEqualTo(ratingsOrderedDown[ratingsOrderedDown.length - 1]);
138     }

164     @Test
165     public void properRankingOfRatings() {
166         final BigDecimal targetShareA = BigDecimal.valueOf(0.001);
167         final BigDecimal targetShareB = targetShareA.multiply(BigDecimal.TEN);
168         final BigDecimal targetShareC = targetShareB.multiply(BigDecimal.TEN);
169 
170         final Map<Rating, StrategyPerRating> strategies = Arrays.stream(Rating.values())
171                 .collect(Collectors.toMap(Function.identity(), r -> Mockito.mock(StrategyPerRating.class)));
172         strategies.put(Rating.A, new StrategyPerRating(Rating.A, targetShareA, 1, 1, 1, 1, BigDecimal.ZERO,
173                 BigDecimal.ONE, 1, 1, true));
174         strategies.put(Rating.B, new StrategyPerRating(Rating.B, targetShareB, 1, 1, 1, 1, BigDecimal.ZERO,
175                 BigDecimal.ONE, 1, 1, true));
176         strategies.put(Rating.C, new StrategyPerRating(Rating.C, targetShareC, 1, 1, 1, 1, BigDecimal.ZERO,
177                 BigDecimal.ONE, 1, 1, true));
178         final SimpleInvestmentStrategy sis = new SimpleInvestmentStrategy(0, Integer.MAX_VALUE, strategies);
179 
180         // all ratings have zero share; C > B > A
181         Map<Rating, BigDecimal> tmp = SimpleInvestmentStategyTest.prepareShareMap(BigDecimal.ZERO, BigDecimal.ZERO,
182                 BigDecimal.ZERO);
183         SimpleInvestmentStategyTest.assertOrder(sis.rankRatingsByDemand(tmp), Rating.C, Rating.B, Rating.A);
184 
185         // A only; B, C overinvested
186         tmp = SimpleInvestmentStategyTest.prepareShareMap(BigDecimal.ZERO, BigDecimal.ONE, BigDecimal.TEN);
187         SimpleInvestmentStategyTest.assertOrder(sis.rankRatingsByDemand(tmp), Rating.A);
188 
189         // B > C > A
190         tmp = SimpleInvestmentStategyTest.prepareShareMap(BigDecimal.valueOf(0.00095), BigDecimal.ZERO,
191                 BigDecimal.valueOf(0.099));
192         SimpleInvestmentStategyTest.assertOrder(sis.rankRatingsByDemand(tmp), Rating.B, Rating.C, Rating.A);
193     }

207     @Test
208     public void doesNotAllowUnderinvesting() {
209         final int balance = 100;
210         final SimpleInvestmentStrategy sis =
211                 new SimpleInvestmentStrategy(balance, 100000, SimpleInvestmentStategyTest.mockStrategies());
212 
213         // all ratings have zero share; C > B > A
214         final Map<Rating, BigDecimal> tmp =
215                 SimpleInvestmentStategyTest.prepareShareMap(BigDecimal.ZERO, BigDecimal.ZERO, BigDecimal.ZERO);
216 
217         // prepare some loans
218         final Loan a1 = SimpleInvestmentStategyTest.mockLoan(1, 2, 3, Rating.A);
219         final Loan b1 = SimpleInvestmentStategyTest.mockLoan(4, 5, 6, Rating.B);
220         final Loan c1 = SimpleInvestmentStategyTest.mockLoan(7, 8, 9, Rating.C);
221         final Loan a2 = SimpleInvestmentStategyTest.mockLoan(10, 11, 12, Rating.A);
222         final List<Loan> loans = Arrays.asList(a1, b1, c1, a2);
223 
224         // make sure we never go below the minimum balance
225         final PortfolioOverview portfolio = Mockito.mock(PortfolioOverview.class);
226         Mockito.when(portfolio.getSharesOnInvestment()).thenReturn(tmp);
227         Mockito.when(portfolio.getCzkInvested()).thenReturn(0);
228         Mockito.when(portfolio.getCzkAvailable()).thenReturn(balance + 1);
229         Assertions.assertThat(sis.getMatchingLoans(loans, portfolio)).isNotEmpty();
230         Mockito.when(portfolio.getCzkAvailable()).thenReturn(balance);
231         Assertions.assertThat(sis.getMatchingLoans(loans, portfolio)).isNotEmpty();
232         Mockito.when(portfolio.getCzkAvailable()).thenReturn(balance - 1);
233         Assertions.assertThat(sis.getMatchingLoans(loans, portfolio)).isEmpty();
234     }

272                 new SimpleInvestmentStrategy(balance, ceiling, SimpleInvestmentStategyTest.mockStrategies());

285
