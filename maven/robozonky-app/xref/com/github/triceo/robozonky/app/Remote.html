45  /**
46   * The base investing algorithm, ready to be executed with {@link ScheduledExecutorService}.
47   */
48  public class Remote implements Callable<Optional<Collection<Investment>>> {
49  
50      private static final Logger LOGGER = LoggerFactory.getLogger(Remote.class);
51      static final Path MARKETPLACE_TIMESTAMP =
52              Paths.get(System.getProperty("user.dir"), "robozonky.lastMarketplaceCheck.timestamp");
53  
54      static Optional<File> storeInvestmentsMade(final Collection<Investment> result, final boolean dryRun) {
55          final String suffix = dryRun ? "dry" : "invested";
56          final LocalDateTime now = LocalDateTime.now();
57          final String filename =
58                  "robozonky." + DateTimeFormatter.ofPattern("yyyyMMddHHmm").format(now) + '.' + suffix;
59          return Remote.storeInvestmentsMade(new File(filename), result);
60      }
61  
62      static Optional<File> storeInvestmentsMade(final File target, final Collection<Investment> result) {
63          if (result.size() == 0) {
64              return Optional.empty();
65          }
66          final Collection<String> output = result.stream()
67                  .map(i -> "#" + i.getLoanId() + ": " + i.getAmount() + " CZK")
68                  .collect(Collectors.toList());
69          try {
70              Files.write(target.toPath(), output);
71              Remote.LOGGER.info("Investments made by RoboZonky during the session were stored in file '{}'.",
72                      target.getAbsolutePath());
73              return Optional.of(target);
74          } catch (final IOException ex) {
75              Remote.LOGGER.warn("Failed writing out the list of investments made in this session.", ex);
76              return Optional.empty();
77          }
78      }
79  
80      static Collection<Loan> getAvailableLoans(final AppContext ctx, final Activity activity) {
81          final boolean shouldSleep = activity.shouldSleep();
82          if (shouldSleep) {
83              Remote.LOGGER.info("RoboZonky is asleep as there is nothing going on.");
84              return Collections.emptyList();
85          } else {
86              Remote.LOGGER.info("Ignoring loans published earlier than {} seconds ago.", ctx.getCaptchaDelayInSeconds());
87              return Collections.unmodifiableList(activity.getAvailableLoans());
88          }
89      }
90  
91      static BigDecimal getAvailableBalance(final AppContext ctx, final ZonkyApi api) {
92          final int dryRunInitialBalance = ctx.getDryRunBalance();
93          return (ctx.isDryRun() && dryRunInitialBalance >= 0) ?
94                  BigDecimal.valueOf(dryRunInitialBalance) : api.getWallet().getAvailableBalance();
95      }
96  
97      static Function<Investor, Collection<Investment>> getInvestingFunction(final AppContext ctx,
98                                                                             final Collection<Loan> availableLoans) {
99          final boolean useStrategy = ctx.getOperatingMode() == OperatingMode.STRATEGY_DRIVEN;
100         // figure out what to execute
101         return useStrategy ? i -> i.invest(ctx.getInvestmentStrategy(), availableLoans) : i -> {
102             final Optional<Investment> optional = i.invest(ctx.getLoanId(), ctx.getLoanAmount());
103             return (optional.isPresent()) ? Collections.singletonList(optional.get()) : Collections.emptyList();
104         };
105     }
106 
107     static Collection<Investment> invest(final AppContext ctx, final ZonkyApi zonky,
108                                          final Collection<Loan> availableLoans) {
109         final BigDecimal balance = Remote.getAvailableBalance(ctx, zonky);
110         final ExecutorService executor = Executors.newWorkStealingPool();
111         final Investor i = new Investor(zonky, balance, executor);
112         final Collection<Investment> result = Remote.getInvestingFunction(ctx, availableLoans).apply(i);
113         executor.shutdownNow();
114         return Collections.unmodifiableCollection(result);
115     }
116 
117     private final AuthenticationHandler auth;
118     private final AppContext ctx;
119 
120     public Remote(final AppContext ctx, final AuthenticationHandler authenticationHandler) {
121         this.ctx = ctx;
122         this.auth = authenticationHandler;
123     }
124 
125     public Optional<Collection<Investment>> call() {
126         final boolean isDryRun = this.ctx.isDryRun();
127         if (isDryRun) {
128             Remote.LOGGER.info("RoboZonky is doing a dry run. It will simulate investing, but not invest any real money.");
129         }
130         final ApiProvider apiProvider = new ApiProvider(); // prepare fresh API factory
131         final Activity activity = new Activity(this.ctx, apiProvider.cache(), Remote.MARKETPLACE_TIMESTAMP);
132         final Collection<Loan> loans = Remote.getAvailableLoans(this.ctx, activity);
133         if (loans.isEmpty()) { // let's fall asleep
134             return Optional.of(Collections.emptyList());
135         }
136         final Optional<Collection<Investment>> optionalResult =
137                 this.auth.execute(apiProvider, api -> Remote.invest(this.ctx, api, loans));
138         activity.settle(); // only settle the marketplace activity when we're sure the app is no longer likely to fail
139         if (optionalResult.isPresent()) {
140             final Collection<Investment> result = optionalResult.get();
141             Remote.storeInvestmentsMade(result, isDryRun);
142             Remote.LOGGER.info("RoboZonky {}invested into {} loans.", isDryRun ? "would have " : "", result.size());
143             return Optional.of(result);
144         } else {
145             return Optional.empty();
146         }
147     }
