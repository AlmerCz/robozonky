43      private static Map<Rating, BigDecimal> prepareShareMap(final BigDecimal ratingA, final BigDecimal ratingB,
44                                                             final BigDecimal ratingC) {
45          final Map<Rating, BigDecimal> map = new EnumMap<>(Rating.class);
46          map.put(Rating.A, ratingA);
47          map.put(Rating.B, ratingB);
48          map.put(Rating.C, ratingC);
49          return Collections.unmodifiableMap(map);
50      }

52      private static void assertOrder(final List<Rating> result, final Rating... ratingsOrderedDown) {
53          Assertions.assertThat(result).hasSize(ratingsOrderedDown.length);
54          if (ratingsOrderedDown.length < 2) {
55              return;
56          } else if (ratingsOrderedDown.length > 3) {
57              throw new IllegalStateException("This should never happen in the test.");
58          }
59          final Rating first = result.get(0);
60          final Rating last = result.get(result.size() - 1);
61          Assertions.assertThat(first).isGreaterThan(last);
62          Assertions.assertThat(first).isEqualTo(ratingsOrderedDown[0]);
63          Assertions.assertThat(last).isEqualTo(ratingsOrderedDown[ratingsOrderedDown.length - 1]);
64      }

66      @Test
67      public void properRankingOfRatings() {
68          final BigDecimal targetShareA = BigDecimal.valueOf(0.001);
69          final BigDecimal targetShareB = targetShareA.multiply(BigDecimal.TEN);
70          final BigDecimal targetShareC = targetShareB.multiply(BigDecimal.TEN);
71  
72          // prepare the mocks of strategy and context
73          final InvestmentStrategy strategy = Mockito.mock(InvestmentStrategy.class);
74          Mockito.when(strategy.getTargetShare(Rating.A)).thenReturn(targetShareA);
75          Mockito.when(strategy.getTargetShare(Rating.B)).thenReturn(targetShareB);
76          Mockito.when(strategy.getTargetShare(Rating.C)).thenReturn(targetShareC);
77          final OperationsContext ctx = Mockito.mock(OperationsContext.class);
78          Mockito.when(ctx.getStrategy()).thenReturn(strategy);
79  
80          // all ratings have zero share; C > B > A
81          Map<Rating, BigDecimal> tmp = OperationsTest.prepareShareMap(BigDecimal.ZERO, BigDecimal.ZERO, BigDecimal.ZERO);
82          OperationsTest.assertOrder(Operations.rankRatingsByDemand(ctx, tmp), Rating.C, Rating.B, Rating.A);
83  
84          // A only; B, C overinvested
85          tmp = OperationsTest.prepareShareMap(BigDecimal.ZERO, BigDecimal.ONE, BigDecimal.TEN);
86          OperationsTest.assertOrder(Operations.rankRatingsByDemand(ctx, tmp), Rating.A);
87  
88          // B > C > A
89          tmp = OperationsTest.prepareShareMap(BigDecimal.valueOf(0.00095), BigDecimal.ZERO, BigDecimal.valueOf(0.099));
90          OperationsTest.assertOrder(Operations.rankRatingsByDemand(ctx, tmp), Rating.B, Rating.C, Rating.A);
91      }

123     private static void assertProperRatingShare(final Map<Rating, BigDecimal> result, final Rating r, final int amount,
124                                                 final int total) {
125         final BigDecimal expectedShare
126                 = BigDecimal.valueOf(amount).divide(BigDecimal.valueOf(total), 4, RoundingMode.HALF_EVEN);
127         Assertions.assertThat(result.get(r)).isEqualTo(expectedShare);

130     @Test
131     public void properRatingShareCalculation() {
132         // mock necessary structures
133         final int amountAA = 300, amountB = 200, amountD = 100;
134         final int totalPie = amountAA + amountB + amountD;
135         final RiskPortfolio riskAA = new RiskPortfolio(Rating.AA, -1, amountAA, -1, -1);
136         final RiskPortfolio riskB = new RiskPortfolio(Rating.B, -1, amountB, -1, -1);
137         final RiskPortfolio riskD = new RiskPortfolio(Rating.D, -1, amountD, -1, -1);
138         final Statistics stats = Mockito.mock(Statistics.class);
139         Mockito.when(stats.getRiskPortfolio()).thenReturn(Arrays.asList(riskAA, riskB, riskD));
140 
141         // check standard operation
142         Map<Rating, BigDecimal> result = Operations.calculateSharesPerRating(stats, Collections.emptyList());
143         OperationsTest.assertProperRatingShare(result, Rating.AA, amountAA, totalPie);
144         OperationsTest.assertProperRatingShare(result, Rating.B, amountB, totalPie);
145         OperationsTest.assertProperRatingShare(result, Rating.D, amountD, totalPie);
146         OperationsTest.assertProperRatingShare(result, Rating.AAAAA, 0, totalPie); // test other ratings included
147         OperationsTest.assertProperRatingShare(result, Rating.AAAA, 0, totalPie);
148         OperationsTest.assertProperRatingShare(result, Rating.AAA, 0, totalPie);
149         OperationsTest.assertProperRatingShare(result, Rating.A, 0, totalPie);
150         OperationsTest.assertProperRatingShare(result, Rating.C, 0, totalPie);
151         // check operation with offline investments
152         final int increment = 200, newTotalPie = totalPie + increment;
153         final List<Investment> investments = OperationsTest.getMockInvestmentWithBalance(increment);
154         final Investment i = investments.get(0);
155         Mockito.when(i.getRating()).thenReturn(Rating.D);
156         result = Operations.calculateSharesPerRating(stats, investments);
157         OperationsTest.assertProperRatingShare(result, Rating.AA, amountAA, newTotalPie);
158         OperationsTest.assertProperRatingShare(result, Rating.B, amountB, newTotalPie);
159         OperationsTest.assertProperRatingShare(result, Rating.D, amountD + increment, newTotalPie);
160         OperationsTest.assertProperRatingShare(result, Rating.AAAAA, 0, newTotalPie); // test other ratings included
161         OperationsTest.assertProperRatingShare(result, Rating.AAAA, 0, newTotalPie);
162         OperationsTest.assertProperRatingShare(result, Rating.AAA, 0, newTotalPie);
163         OperationsTest.assertProperRatingShare(result, Rating.A, 0, newTotalPie);
164         OperationsTest.assertProperRatingShare(result, Rating.C, 0, newTotalPie);
165     }

167     @Test
168     public void properLogin() {
169         final Authentication tmp = Mockito.mock(Authentication.class);
170         Mockito.when(tmp.getZonkyApi()).thenReturn(Mockito.mock(ZonkyApi.class));
171         Mockito.when(tmp.getZonkyApiToken()).thenReturn(Mockito.mock(ZonkyApiToken.class));
172         final Authenticator auth = Mockito.mock(Authenticator.class);
173         Mockito.when(auth.authenticate(Mockito.any(), Mockito.any(), Mockito.any(), Mockito.any())).thenReturn(tmp);
174         final Optional<Operations.LoginResult> optional = Operations.login(auth, true, 1000);
175         Assertions.assertThat(optional).isPresent();
176         final Operations.LoginResult c = optional.get();
177         Assertions.assertThat(c.getZonkyApiToken()).isSameAs(tmp.getZonkyApiToken());
178         Assertions.assertThat(c.getOperationsContext()).isNotNull();
179         Assertions.assertThat(c.getOperationsContext().getZonkyApi()).isSameAs(tmp.getZonkyApi());
180         Operations.logout(c.getOperationsContext());
181         Mockito.verify(tmp.getZonkyApi(), Mockito.times(1)).logout();
182     }
