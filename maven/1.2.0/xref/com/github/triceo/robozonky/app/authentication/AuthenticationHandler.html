17  package com.github.triceo.robozonky.app.authentication;
18  
19  import java.io.Reader;
20  import java.io.StringReader;
21  import java.time.Duration;
22  import java.time.LocalDateTime;
23  import java.time.temporal.ChronoUnit;
24  import java.time.temporal.TemporalUnit;
25  import java.util.Optional;
26  import javax.xml.bind.JAXBException;
27  
28  import com.github.triceo.robozonky.authentication.Authenticator;
29  import com.github.triceo.robozonky.remote.ZonkyApiToken;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  public class AuthenticationHandler {
34  
35      private static final Logger LOGGER = LoggerFactory.getLogger(AuthenticationHandler.class);
36  
37      /**
38       * Build authentication mechanism that will keep the session alive via the use of session token. The mechanism will
39       * never log out, but the session may expire if not refresh regularly. This is potentially unsafe, as it will
40       * eventually store a plain-text access token on the hard drive, for everyone to see.
41       *
42       * @param data Provider for the sensitive information, such as passwords and tokens.
43       * @return The desired authentication method.
44       */
45      public static AuthenticationHandler tokenBased(final SensitiveInformationProvider data) {
46          return new AuthenticationHandler(data, true);
47      }
48  
49      /**
50       * Build authentication mechanism that will log out at the end of RoboZonky's operations. This will ignore the
51       * access tokens.
52       *
53       * @param data Provider for the sensitive information, such as passwords and tokens.
54       * @return The desired authentication method.
55       */
56      public static AuthenticationHandler passwordBased(final SensitiveInformationProvider data) {
57          return new AuthenticationHandler(data, false);
58      }
59  
60      private final boolean tokenBased;
61      private final SensitiveInformationProvider data;
62      private long tokenRefreshBeforeExpirationInSeconds = 60;
63  
64      private AuthenticationHandler(final SensitiveInformationProvider data, final boolean tokenBased) {
65          this.data = data;
66          this.tokenBased = tokenBased;
67      }
68  
69      /**
70       * Optionally provide the earliest time before the token expiration at which the access token should be refreshed.
71       * The token will only be refreshed if RoboZonky is launched some time between then and the token expiration.
72       *
73       * If this method is not called, the default value is {@link #tokenRefreshBeforeExpirationInSeconds} seconds.
74       *
75       * @param time Access token will be refreshed after expiration minus this.
76       * @param unit Unit of time applied to the previous argument.
77       * @return This.
78       */
79      public AuthenticationHandler withTokenRefreshingBeforeExpiration(final long time, final TemporalUnit unit) {
80          this.tokenRefreshBeforeExpirationInSeconds = Duration.of(time, unit).getSeconds();
81          return this;
82      }
83  
84      private Authenticator buildWithPassword() {
85          return Authenticator.withCredentials(this.data.getUsername(), this.data.getPassword());
86      }
87  
88      /**
89       * Based on information received until this point, decide on the proper authentication method.
90       *
91       * @return Authentication method matching user preferences.
92       */
93      public Authenticator build() {
94          final Optional<Reader> tokenStream = this.data.getToken();
95          if (!this.tokenBased) {
96              AuthenticationHandler.LOGGER.debug("Password-based authentication requested.");
97              if (!this.data.setToken()) {
98                  AuthenticationHandler.LOGGER.info("Failed to delete stale access token.");
99              }
100             return this.buildWithPassword();
101         } else if (!tokenStream.isPresent()) { // no token available, also using password-based
102             AuthenticationHandler.LOGGER.debug("Token file not available, using password-based authentication.");
103             return this.buildWithPassword();
104         }
105         boolean deleteToken = false;
106         try {
107             final ZonkyApiToken token = ZonkyApiToken.unmarshal(tokenStream.get());
108             final LocalDateTime obtained = this.data.getTokenSetDate().get();
109             final LocalDateTime expires = obtained.plus(token.getExpiresIn(), ChronoUnit.SECONDS);
110             AuthenticationHandler.LOGGER.debug("Token obtained on {}, expires on {}.", obtained, expires);
111             final LocalDateTime now = LocalDateTime.now();
112             if (expires.isBefore(now)) {
113                 AuthenticationHandler.LOGGER.debug("Token {} expired, using password-based authentication.",
114                         token.getAccessToken());
115                 deleteToken = true;
116                 return this.buildWithPassword();
117             }
118             if (expires.minus(this.tokenRefreshBeforeExpirationInSeconds, ChronoUnit.SECONDS).isBefore(now)) {
119                 AuthenticationHandler.LOGGER.debug("Access token expiring, will be refreshed.");
120                 deleteToken = true;
121                 return Authenticator.withAccessTokenAndRefresh(this.data.getUsername(), token);
122             } else {
123                 AuthenticationHandler.LOGGER.debug("Reusing access token.");
124                 return Authenticator.withAccessToken(this.data.getUsername(), token);
125             }
126         } catch (final JAXBException ex) {
127             AuthenticationHandler.LOGGER.warn("Failed parsing token, using password-based authentication.", ex);
128             deleteToken = true;
129             return this.buildWithPassword();
130         } finally {
131             if (deleteToken && !this.data.setToken()) {
132                 AuthenticationHandler.LOGGER.warn("Failed deleting token, authentication may stop working.");
133             }
134         }
135     }
136 
137     /**
138      * Decide whether or not to log out, based on user preferences.
139      *
140      * @param token Token in question.
141      * @return True if RoboZonky should log out, false otherwise.
142      */
143     public boolean processToken(final ZonkyApiToken token) {
144         if (!this.tokenBased) { // not using token; always logout
145             return true;
146         }
147         final Optional<Reader> tokenStream = this.data.getToken();
148         if (tokenStream.isPresent()) { // token already exists, do not logout
149             return false;
150         } else try { // try to store token
151             final String marshalled = ZonkyApiToken.marshal(token);
152             final boolean tokenStored = this.data.setToken(new StringReader(marshalled));
153             if (tokenStored) {
154                 AuthenticationHandler.LOGGER.debug("Token stored successfully.");
155                 return false;
156             } else {
157                 AuthenticationHandler.LOGGER.debug("Failed storing token.");
158                 return true;
159             }
160         } catch (final JAXBException ex) {
161             AuthenticationHandler.LOGGER.info("Failed writing access token, will need to use password next time.", ex);
162             return true;
163         }
164     }
