16  package com.github.triceo.robozonky.strategy;
17  
18  import java.math.BigDecimal;
19  
20  import com.github.triceo.robozonky.remote.Loan;
21  import com.github.triceo.robozonky.remote.Rating;
22  import org.assertj.core.api.Assertions;
23  import org.junit.Test;
24  import org.mockito.Mockito;
25  
26  public class StrategyPerRatingTest {
27  
28      private static final Rating DIFFERENT_RATING = Rating.B;
29      private static final int TESTED_TERM_LENGTH = 2;
30      private static final int MAXIMUM_LOAN_INVESTMENT = 200;
31      private static final BigDecimal MAXIMUM_LOAN_SHARE = BigDecimal.valueOf(0.01);
32      private static final StrategyPerRating STRATEGY = new StrategyPerRating(Rating.A, BigDecimal.valueOf(0.15),
33              StrategyPerRatingTest.TESTED_TERM_LENGTH - 1, -1, StrategyPerRatingTest.MAXIMUM_LOAN_INVESTMENT,
34              StrategyPerRatingTest.MAXIMUM_LOAN_SHARE, true);
35  
36      @Test(expected = IllegalArgumentException.class)
37      public void loanIsNotAcceptableWithoutMatchingRating() {
38          final Loan mockLoan = Mockito.mock(Loan.class);
39          Mockito.when(mockLoan.getRating()).thenReturn(StrategyPerRatingTest.DIFFERENT_RATING);
40  
41          StrategyPerRatingTest.STRATEGY.isAcceptable(mockLoan);
42      }
43  
44      @Test(expected = IllegalArgumentException.class)
45      public void loanIsNotAcceptableWithoutMatchingRating2() {
46          final Loan mockLoan = Mockito.mock(Loan.class);
47          Mockito.when(mockLoan.getRating()).thenReturn(StrategyPerRatingTest.DIFFERENT_RATING);
48  
49          StrategyPerRatingTest.STRATEGY.recommendInvestmentAmount(mockLoan);
50      }
51  
52      @Test
53      public void loanIsNotAcceptableDueToTermMismatch() {
54          final Loan mockLoan = Mockito.mock(Loan.class);
55          Mockito.when(mockLoan.getRating()).thenReturn(StrategyPerRatingTest.STRATEGY.getRating());
56  
57          // term length within limits
58          Mockito.when(mockLoan.getTermInMonths()).thenReturn(StrategyPerRatingTest.TESTED_TERM_LENGTH);
59          Assertions.assertThat(StrategyPerRatingTest.STRATEGY.isAcceptable(mockLoan)).isTrue();
60  
61          // less than minimal term length
62          Mockito.when(mockLoan.getTermInMonths()).thenReturn(0);
63          Assertions.assertThat(StrategyPerRatingTest.STRATEGY.isAcceptable(mockLoan)).isFalse();
64      }
65  
66      private void overInvestmentScenario(double loanAmount) {
67          final Loan mockLoan = Mockito.mock(Loan.class);
68          Mockito.when(mockLoan.getRating()).thenReturn(StrategyPerRatingTest.STRATEGY.getRating());
69          Mockito.when(mockLoan.getAmount()).thenReturn(loanAmount);
70  
71          final int recommendedInvestment = StrategyPerRatingTest.STRATEGY.recommendInvestmentAmount(mockLoan);
72          Assertions.assertThat(recommendedInvestment).isLessThanOrEqualTo(StrategyPerRatingTest.MAXIMUM_LOAN_INVESTMENT);
73          final BigDecimal share = BigDecimal.valueOf(recommendedInvestment / mockLoan.getAmount());
74          Assertions.assertThat(share).isLessThanOrEqualTo(StrategyPerRatingTest.MAXIMUM_LOAN_SHARE);
75      }
76  
77      @Test
78      public void loanIsOnlyAcceptableWhenNotOverInvested() {
79          this.overInvestmentScenario(StrategyPerRatingTest.MAXIMUM_LOAN_INVESTMENT);
80          this.overInvestmentScenario(StrategyPerRatingTest.MAXIMUM_LOAN_INVESTMENT * 2);
81          this.overInvestmentScenario(StrategyPerRatingTest.MAXIMUM_LOAN_INVESTMENT / 2);
82      }
83  
84  }
