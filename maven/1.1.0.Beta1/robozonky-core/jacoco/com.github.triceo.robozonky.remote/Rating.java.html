package com.github.triceo.robozonky.remote;

public enum Rating {

    AAAAA("A**"),
    AAAA("A*"),
    AAA("A++"),
    AA("A+"),
    A("A"),
    B("B"),
    C("C"),
    D("D");


    private final String code;

    Rating(final String code) {
        this.code = code;
    }

    public String getCode() {
        return code;
    }

}
