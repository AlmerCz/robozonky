17  package com.github.triceo.robozonky.strategy;
18  
19  import java.math.BigDecimal;
20  import java.util.EnumMap;
21  import java.util.Map;
22  
23  import com.github.triceo.robozonky.remote.Rating;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  public class StrategyBuilder {
28  
29      private static final Logger LOGGER = LoggerFactory.getLogger(StrategyBuilder.class);
30  
31      private final Map<Rating, StrategyPerRating> individualStrategies = new EnumMap<>(Rating.class);
32  
33      public StrategyBuilder addIndividualStrategy(final Rating r, final BigDecimal targetShare, final int minTerm,
34                                                   final int maxTerm, final int maxLoanAmount,
35                                                   final BigDecimal maxLoanShare, final boolean preferLongerTerms) {
36          if (individualStrategies.containsKey(r)) {
37              throw new IllegalArgumentException("Already added strategy for rating " + r);
38          }
39          individualStrategies.put(r,
40                  new StrategyPerRating(r, targetShare, minTerm, maxTerm, maxLoanAmount, maxLoanShare, preferLongerTerms));
41          StrategyBuilder.LOGGER.debug("Adding strategy for rating '{}'.", r.getCode());
42          StrategyBuilder.LOGGER.debug("Target share for rating '{}' among total investments is {}.", r.getCode(), targetShare);
43          StrategyBuilder.LOGGER.debug("Range of acceptable investment terms for rating '{}' is <{}, {}> months.", r.getCode(),
44                  minTerm == -1 ? 0 : minTerm, maxTerm == -1 ? "+inf" : maxTerm);
45          StrategyBuilder.LOGGER.debug("Maximum investment amount for rating '{}' is {} CZK.", r.getCode(), maxLoanAmount);
46          StrategyBuilder.LOGGER.debug("Maximum investment share for rating '{}' is {}.", r.getCode(), maxLoanShare);
47          StrategyBuilder.LOGGER.debug("Rating '{}' will prefer longer terms.", r.getCode());
48          return this;
49      }
50  
51      public InvestmentStrategy build() {
52          if (individualStrategies.size() != Rating.values().length) {
53              throw new IllegalStateException("Strategy is incomplete.");
54          }
55          return new InvestmentStrategy(individualStrategies);
56      }
57  
58  }
