package com.github.triceo.robozonky.app;

public enum ReturnCode {
    OK(0), ERROR_WRONG_PARAMETERS(1), ERROR_LOGIN(2), ERROR_UNEXPECTED(255);

    private final int code;

    ReturnCode(int code) {
        this.code = code;
    }

    public int getCode() {
        return code;
    }
}
